local function startFly()
    local char = LocalPlayer.Character
    if not char then return end
    
    local humanoid = char:FindFirstChild("Humanoid")
    if humanoid then
        humanoid.PlatformStand = true
    end
    
    local rootPart = char:FindFirstChild("HumanoidRootPart")
    if not rootPart then return end
    
    bodyVelocity = Instance.new("BodyVelocity")
    bodyVelocity.MaxForce = Vector3.new(1, 1, 1) * 100000
    bodyVelocity.Parent = rootPart
    
    bodyGyro = Instance.new("BodyGyro")
    bodyGyro.MaxTorque = Vector3.new(1, 1, 1) * 100000
    bodyGyro.Parent = rootPart
    
    local function updateFly()
        if not flyEnabled or not char or not char.Parent then return end
        
        local moveDirection = Vector3.new(
            (UserInputService:IsKeyDown(Enum.KeyCode.D) and 1 or 0) - (UserInputService:IsKeyDown(Enum.KeyCode.A) and 1 or 0),
            (UserInputService:IsKeyDown(Enum.KeyCode.Space) and 1 or 0) - (UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) and 1 or 0),
            (UserInputService:IsKeyDown(Enum.KeyCode.S) and 1 or 0) - (UserInputService:IsKeyDown(Enum.KeyCode.W) and 1 or 0)
        )
        
        local cameraCFrame = Camera.CFrame
        local moveVelocity = cameraCFrame:VectorToWorldSpace(moveDirection) * flySpeed
        
        if bodyVelocity then
            bodyVelocity.Velocity = moveVelocity
        end
        
        local lookDirection = Camera.CFrame.LookVector
        if bodyGyro then
            bodyGyro.CFrame = CFrame.new(rootPart.Position, rootPart.Position + lookDirection)
        end
    end
    
    flyConnection = RunService.RenderStepped:Connect(updateFly)
end

local function stopFly()
    if flyConnection then
        flyConnection:Disconnect()
        flyConnection = nil
    end
    
    local char = LocalPlayer.Character
    if char then
        local humanoid = char:FindFirstChild("Humanoid")
        if humanoid then
            humanoid.PlatformStand = false
        end
        
        if bodyVelocity then
            bodyVelocity:Destroy()
            bodyVelocity = nil
        end
        
        if bodyGyro then
            bodyGyro:Destroy()
            bodyGyro = nil
        end
    end
end

local function toggleFly()
    flyEnabled = not flyEnabled
    
    if flyEnabled then
        startFly()
    else
        stopFly()
    end
end

local function applyHitbox(player)
    if not hitboxEnabled then return end
    
    local character = player.Character
    if not character then return end
    
    local head = character:FindFirstChild("Head")
    local torso = character:FindFirstChild("HumanoidRootPart")
    
    if head and not head:FindFirstChild("HitboxExpander") then
        local headSize = Instance.new("NumberValue")
        headSize.Name = "HitboxExpander"
        headSize.Value = hitboxHeadSize
        headSize.Parent = head
        
        local headConn = head:GetPropertyChangedSignal("Size"):Connect(function()
            if hitboxEnabled and head:FindFirstChild("HitboxExpander") then
                local scale = head.HitboxExpander.Value
                head.Size = Vector3.new(2, 2, 2) * scale
            end
        end)
        head.Size = Vector3.new(2, 2, 2) * hitboxHeadSize
        
        if not hitboxConnections[player] then hitboxConnections[player] = {} end
        hitboxConnections[player].head = headConn
    end
    
    if torso and not torso:FindFirstChild("HitboxExpander") then
        local torsoSize = Instance.new("NumberValue")
        torsoSize.Name = "HitboxExpander"
        torsoSize.Value = hitboxBodySize
        torsoSize.Parent = torso
        
        local torsoConn = torso:GetPropertyChangedSignal("Size"):Connect(function()
            if hitboxEnabled and torso:FindFirstChild("HitboxExpander") then
                local scale = torso.HitboxExpander.Value
                torso.Size = Vector3.new(2, 3, 1) * scale
            end
        end)
        torso.Size = Vector3.new(2, 3, 1) * hitboxBodySize
        
        if not hitboxConnections[player] then hitboxConnections[player] = {} end
        hitboxConnections[player].torso = torsoConn
    end
end

local function removeHitbox(player)
    local character = player.Character
    if character then
        local head = character:FindFirstChild("Head")
        local torso = character:FindFirstChild("HumanoidRootPart")
        
        if head then
            local sizeVal = head:FindFirstChild("HitboxExpander")
            if sizeVal then sizeVal:Destroy() end
            head.Size = Vector3.new(2, 2, 2)
        end
        
        if torso then
            local sizeVal = torso:FindFirstChild("HitboxExpander")
            if sizeVal then sizeVal:Destroy() end
            torso.Size = Vector3.new(2, 3, 1)
        end
    end
    
    if hitboxConnections[player] then
        if hitboxConnections[player].head then hitboxConnections[player].head:Disconnect() end
        if hitboxConnections[player].torso then hitboxConnections[player].torso:Disconnect() end
        hitboxConnections[player] = nil
    end
end

local function toggleHitbox()
    hitboxEnabled = not hitboxEnabled
    
    if hitboxEnabled then
        for _, player in ipairs(Players:GetPlayers()) do
            if player ~= LocalPlayer then
                applyHitbox(player)
            end
        end
        
        Players.PlayerAdded:Connect(function(player)
            if hitboxEnabled and player ~= LocalPlayer then
                player.CharacterAdded:Connect(function()
                    task.wait(0.5)
                    if hitboxEnabled and player ~= LocalPlayer then
                        applyHitbox(player)
                    end
                end)
            end
        end)
        
        Players.PlayerRemoving:Connect(function(player)
            removeHitbox(player)
        end)
    else
        for _, player in ipairs(Players:GetPlayers()) do
            if player ~= LocalPlayer then
                removeHitbox(player)
            end
        end
    end
end

local function updateHitboxSizes()
    if not hitboxEnabled then return end
    
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            local character = player.Character
            if character then
                local head = character:FindFirstChild("Head")
                local torso = character:FindFirstChild("HumanoidRootPart")
                
                if head and head:FindFirstChild("HitboxExpander") then
                    head.HitboxExpander.Value = hitboxHeadSize
                    head.Size = Vector3.new(2, 2, 2) * hitboxHeadSize
                end
                
                if torso and torso:FindFirstChild("HitboxExpander") then
                    torso.HitboxExpander.Value = hitboxBodySize
                    torso.Size = Vector3.new(2, 3, 1) * hitboxBodySize
                end
            end
        end
    end
end

local function createESP(player)
    if espObjects[player] then return end
    
    local highlight = Instance.new("Highlight")
    highlight.Name = "Highlight_" .. player.Name
    highlight.FillColor = Color3.fromRGB(255, 0, 0)
    highlight.FillTransparency = 0.5
    highlight.OutlineColor = Color3.fromRGB(255, 0, 0)
    highlight.OutlineTransparency = 0.3
    highlight.Adornee = player.Character
    highlight.Parent = player.Character
    
    local billboard = Instance.new("BillboardGui")
    billboard.Name = "ESP_" .. player.Name
    billboard.Size = UDim2.new(0, 100, 0, 40)
    billboard.StudsOffset = Vector3.new(0, 2.5, 0)
    billboard.AlwaysOnTop = true
    billboard.Parent = player.Character and player.Character:FindFirstChild("Head")
    
    local healthText = Instance.new("TextLabel")
    healthText.Size = UDim2.new(1, 0, 1, 0)
    healthText.BackgroundTransparency = 1
    healthText.TextColor3 = Color3.fromRGB(255, 255, 255)
    healthText.TextStrokeTransparency = 0.3
    healthText.TextSize = 14
    healthText.Font = Enum.Font.GothamBold
    healthText.Text = "100 HP"
    healthText.Parent = billboard
    
    local nameText = Instance.new("TextLabel")
    nameText.Size = UDim2.new(1, 0, 0.4, 0)
    nameText.Position = UDim2.new(0, 0, 1, 0)
    nameText.BackgroundTransparency = 1
    nameText.Text = player.Name
    nameText.TextColor3 = Color3.fromRGB(200, 200, 200)
    nameText.TextSize = 10
    nameText.Font = Enum.Font.Gotham
    nameText.Parent = billboard
    
    espObjects[player] = {
        highlight = highlight,
        billboard = billboard,
        healthText = healthText
    }
    
    local function updateHealth()
        if player.Character and player.Character:FindFirstChild("Humanoid") then
            local humanoid = player.Character.Humanoid
            local health = humanoid.Health
            local maxHealth = humanoid.MaxHealth
            healthText.Text = math.floor(health) .. "/" .. math.floor(maxHealth) .. " HP"
            
            if health / maxHealth > 0.6 then
                healthText.TextColor3 = Color3.fromRGB(0, 255, 0)
            elseif health / maxHealth > 0.3 then
                healthText.TextColor3 = Color3.fromRGB(255, 255, 0)
            else
                healthText.TextColor3 = Color3.fromRGB(255, 0, 0)
            end
        end
    end
    
    updateHealth()
    
    local connection
    connection = RunService.Heartbeat:Connect(function()
        if not player.Character or not player.Character:FindFirstChild("Head") then
            connection:Disconnect()
            return
        end
        updateHealth()
        
        if billboard and player.Character and player.Character:FindFirstChild("Head") then
            billboard.Parent = player.Character.Head
        end
        if highlight and player.Character then
            highlight.Adornee = player.Character
        end
    end)
    
    espObjects[player].updateConnection = connection
end

local function removeESP(player)
    local data = espObjects[player]
    if data then
        if data.highlight then data.highlight:Destroy() end
        if data.billboard then data.billboard:Destroy() end
        if data.updateConnection then data.updateConnection:Disconnect() end
        espObjects[player] = nil
    end
end

local function toggleESP()
    espEnabled = not espEnabled
    
    if espEnabled then
        for _, player in ipairs(Players:GetPlayers()) do
            if player ~= LocalPlayer and player.Character then
                createESP(player)
            end
        end
        
        Players.PlayerAdded:Connect(function(player)
            if espEnabled then
                player.CharacterAdded:Connect(function()
                    task.wait(1)
                    if espEnabled and player ~= LocalPlayer and player.Character then
                        createESP(player)
                    end
                end)
            end
        end)
        
        Players.PlayerRemoving:Connect(function(player)
            removeESP(player)
        end)
    else
        for player, _ in pairs(espObjects) do
            removeESP(player)
        end
    end
end

local function startAim()
    if aimConnection then aimConnection:Disconnect() end
    aimConnection = RunService.RenderStepped:Connect(function()
        if not aimEnabled then return end
        
        local currentTarget = aimTarget
        if aimCycleEnabled then
            currentTarget = aimCycleCurrent
        end
        
        local closestPlayer = nil
        local closestDistance = math.huge
        local screenCenter = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
        
        for _, player in ipairs(Players:GetPlayers()) do
            if player ~= LocalPlayer and player.Character and player.Character:FindFirstChild(currentTarget) and player.Character:FindFirstChild("Humanoid") and player.Character.Humanoid.Health > 0 then
                local targetPart = player.Character:FindFirstChild(currentTarget)
                if targetPart then
                    local screenPoint, onScreen = Camera:WorldToScreenPoint(targetPart.Position)
                    
                    if onScreen then
                        local distance = (Vector2.new(screenPoint.X, screenPoint.Y) - screenCenter).Magnitude
                        if distance <= aimFOV and distance < closestDistance then
                            closestDistance = distance
                            closestPlayer = player
                        end
                    end
                end
            end
        end
        
        if closestPlayer and closestPlayer.Character and closestPlayer.Character:FindFirstChild(currentTarget) then
            local targetPos = closestPlayer.Character:FindFirstChild(currentTarget).Position
            local cameraPos = Camera.CFrame.Position
            local targetCFrame = CFrame.new(cameraPos, targetPos)
            
            if aimSmoothness > 0 then
                currentCameraCFrame = currentCameraCFrame:Lerp(targetCFrame, aimSmoothness)
                Camera.CFrame = currentCameraCFrame
            else
                Camera.CFrame = targetCFrame
                currentCameraCFrame = targetCFrame
            end
        end
    end)
end

local function toggleAim()
    aimEnabled = not aimEnabled
    aimCircle.Visible = aimEnabled
    aimGlow.Visible = aimEnabled
    
    if aimEnabled then
        currentCameraCFrame = Camera.CFrame
        startAim()
        if aimCycleEnabled then
            aimCycleTimer = 0
            aimCycleCurrent = aimTarget
            startCycle()
        end
    else
        if aimConnection then
            aimConnection:Disconnect()
            aimConnection = nil
        end
    end
end

local cycleConnection
local function startCycle()
    if cycleConnection then cycleConnection:Disconnect() end
    cycleConnection = RunService.Heartbeat:Connect(function(dt)
        if aimEnabled and aimCycleEnabled then
            aimCycleTimer = aimCycleTimer + dt
            if aimCycleCurrent == "Head" and aimCycleTimer >= aimCycleHeadTime then
                aimCycleCurrent = "HumanoidRootPart"
                aimCycleTimer = 0
            elseif aimCycleCurrent == "HumanoidRootPart" and aimCycleTimer >= aimCycleBodyTime then
                aimCycleCurrent = "Head"
                aimCycleTimer = 0
            end
        end
    end)
end

local function toggleCycle()
    aimCycleEnabled = not aimCycleEnabled
    if aimCycleEnabled then
        aimCycleTimer = 0
        aimCycleCurrent = aimTarget
        startCycle()
    end
end

local function ShowMenuWithAnimation()
    if isAnimating or menuVisible then return end
    isAnimating = true
    menuVisible = true
    mainFrame.Visible = true
    
    mainFrame.Size = UDim2.new(0, 0, 0, 0)
    mainFrame.Position = UDim2.new(0.5, 0, 0.5, 0)
    
    local tweenSize = TweenService:Create(mainFrame, TweenInfo.new(0.35, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
        Size = UDim2.new(0, 560, 0, 990)
    })
    local tweenPos = TweenService:Create(mainFrame, TweenInfo.new(0.35, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
        Position = UDim2.new(0.5, -280, 0.5, -495)
    })
    tweenSize:Play()
    tweenPos:Play()
    tweenPos.Completed:Connect(function()
        isAnimating = false
    end)
end

local function HideMenuWithAnimation()
    if isAnimating or not menuVisible then return end
    isAnimating = true
    menuVisible = false
    
    local tweenSize = TweenService:Create(mainFrame, TweenInfo.new(0.35, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {
        Size = UDim2.new(0, 0, 0, 0)
    })
    local tweenPos = TweenService:Create(mainFrame, TweenInfo.new(0.35, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {
        Position = UDim2.new(0.5, 0, 0.5, 0)
    })
    tweenSize:Play()
    tweenPos:Play()
    tweenPos.Completed:Connect(function()
        mainFrame.Visible = false
        isAnimating = false
    end)
end

UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode.Name == teleportKey and teleportEnabled then
        local mousePos = Mouse.Hit.Position
        if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
            LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(mousePos.X, mousePos.Y + 3, mousePos.Z)
        end
    end
    
    if input.KeyCode.Name == aimKey then
        if aimMode == "Hold" then
            if not aimHoldActive then
                aimHoldActive = true
                aimEnabled = true
                aimCircle.Visible = true
                aimGlow.Visible = true
                currentCameraCFrame = Camera.CFrame
                startAim()
                if aimCycleEnabled then
                    aimCycleTimer = 0
                    aimCycleCurrent = aimTarget
                    startCycle()
                end
                btn2.Text = "Аим (наводка на человека) [ВКЛ]"
                btn2.TextColor3 = Color3.fromRGB(100, 255, 100)
            end
        elseif aimMode == "Toggle" then
            toggleAim()
            if aimEnabled then
                if aimCycleEnabled then
                    aimCycleTimer = 0
                    aimCycleCurrent = aimTarget
                    startCycle()
                end
                btn2.Text = "Аим (наводка на человека) [ВКЛ]"
                btn2.TextColor3 = Color3.fromRGB(100, 255, 100)
            else
                btn2.Text = "Аим (наводка на человека)"
                btn2.TextColor3 = Color3.fromRGB(200, 200, 220)
            end
        end
    end
    
    if input.KeyCode.Name == espKey then
        toggleESP()
        if espEnabled then
            btn3.Text = "ESP (видение за стенами) [ВКЛ]"
            btn3.TextColor3 = Color3.fromRGB(100, 255, 100)
        else
            btn3.Text = "ESP (видение за стенами)"
            btn3.TextColor3 = Color3.fromRGB(200, 200, 220)
        end
    end
    
    if input.KeyCode.Name == flyKey then
        toggleFly()
        if flyEnabled then
            local flyBtn = flyContent:FindFirstChildWhichIsA("TextButton")
            if flyBtn and flyBtn.Name == "enableBtn" then
                flyBtn.Text = "Полёт: ВКЛ"
                flyBtn.BackgroundColor3 = Color3.fromRGB(80, 120, 255)
            end
        else
            local flyBtn = flyContent:FindFirstChildWhichIsA("TextButton")
            if flyBtn and flyBtn.Name == "enableBtn" then
                flyBtn.Text = "Полёт: ВЫКЛ"
                flyBtn.BackgroundColor3 = Color3.fromRGB(45, 45, 60)
            end
        end
    end
    
    if input.KeyCode.Name == menuKey then
        if menuVisible then
            HideMenuWithAnimation()
        else
            ShowMenuWithAnimation()
        end
    end
end)

UserInputService.InputEnded:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode.Name == aimKey and aimMode == "Hold" then
        if aimHoldActive then
            aimHoldActive = false
            aimEnabled = false
            aimCircle.Visible = false
            aimGlow.Visible = false
            if aimConnection then
                aimConnection:Disconnect()
                aimConnection = nil
            end
            btn2.Text = "Аим (наводка на человека)"
            btn2.TextColor3 = Color3.fromRGB(200, 200, 220)
        end
    end
end)

local dragging = false
local dragStartPos
local frameStartPos

topPanel.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStartPos = input.Position
        frameStartPos = mainFrame.Position
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - dragStartPos
        mainFrame.Position = UDim2.new(frameStartPos.X.Scale, frameStartPos.X.Offset + delta.X, frameStartPos.Y.Scale, frameStartPos.Y.Offset + delta.Y)
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)
